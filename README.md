
## 安装
```shell
npm install manyi-ui -S
```

## 使用教程
``` javascript
import Vue from 'vue'
import ManyiUI from 'manyi-ui/src'

Vue.use(ManyiUI)

// or
import {
  Select,
  Button
  // ...
} from 'manyi-ui'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
